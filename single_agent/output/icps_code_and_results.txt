```
# Python Code Used
import pandas as pd
import matplotlib.pyplot as plt

# Load CSV data
df = pd.read_csv('csv_data.csv')

# Filter for orders in CAD
df_cad = df[df['currency'] == 'CAD']

# Calculate total values of each order
df_cad['total_value'] = df_cad['quantity_ordered'] * df_cad['unit_price']

# Group by po_number and sum total values
bar_data = df_cad.groupby('po_number')['total_value'].sum()

# Plotting
total_orders = bar_data.index
total_values = bar_data.values
plt.bar(total_orders, total_values, color='blue')
plt.xlabel('Purchase Order Number')
plt.ylabel('Total Order Value (CAD)')
plt.title('Total Order Values in CAD')
plt.xticks(rotation=45)
plt.tight_layout()
plt.savefig('total_order_values_cad.png')
plt.show()
```
